Bash - автоматизация задач

1.!/bin/bash
    
    / Пример скрипта резервного копирования
    
    2. Переменные
    SOURCE_DIR="/path/to/source"
    BACKUP_DIR="/path/to/backup"
    DATE=$(date +%Y-%m-%d)
    BACKUP_FILE="backup-$DATE.tar.gz"
    
    3. Создание архива
    echo "Creating backup of $SOURCE_DIR..."
    tar -czf "$BACKUP_DIR/$BACKUP_FILE" "$SOURCE_DIR"
    
    4. Проверка результата
    if [ $? -eq 0 ]; then
        echo "Backup completed successfully: $BACKUP_DIR/$BACKUP_FILE"
    else
        echo "Backup failed!"
        exit 1
    fi
    
    5. Удаление старых резервных копий (старше 30 дней)
    find "$BACKUP_DIR" -name "backup-*.tar.gz" -mtime +30 -delete
    echo "Old backups cleaned up."

Настройка сетевых интерфейсов

1. Linux - временная настройка
ip addr add 192.168.1.100/24 dev eth0
ip route add default via 192.168.1.1

2. Linux - постоянная настройка (Debian/Ubuntu)
nano/mcedit/vim: /etc/network/interfaces
auto eth0
iface eth0 inet static
    address 192.168.1.100
    netmask 255.255.255.0
    gateway 192.168.1.1
    dns-nameservers 8.8.8.8 8.8.4.4

3. Windows - PowerShell
New-NetIPAddress -InterfaceAlias "Ethernet" -IPAddress 192.168.1.100 -PrefixLength 24 -DefaultGateway 192.168.1.1
Set-DnsClientServerAddress -InterfaceAlias "Ethernet" -ServerAddresses 8.8.8.8,8.8.4.4

Настройка брандмауэра

1. Linux - iptables
iptables -A INPUT -p tcp --dport 22 -j ACCEPT
iptables -A INPUT -p tcp --dport 80 -j ACCEPT
iptables -A INPUT -m state --state ESTABLISHED,RELATED -j ACCEPT
iptables -A INPUT -j DROP

2. Linux - firewalld
firewall-cmd --permanent --add-service=ssh
firewall-cmd --permanent --add-service=http
firewall-cmd --reload

3. Windows - PowerShell
New-NetFirewallRule -DisplayName "Allow SSH" -Direction Inbound -Protocol TCP -LocalPort 22 -Action Allow
New-NetFirewallRule -DisplayName "Allow HTTP" -Direction Inbound -Protocol TCP -LocalPort 80 -Action Allow

Базы данных

MySQL / MariaDB

-- Подключение к серверу
mysql -u username -p

-- Создание базы данных
CREATE DATABASE mydatabase;
USE mydatabase;

-- Создание таблицы
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Вставка данных
INSERT INTO users (username, email) VALUES ('john_doe', 'john@example.com');

-- Выборка данных с JOIN
SELECT u.username, p.title 
FROM users u
JOIN posts p ON u.id = p.user_id
WHERE u.username = 'john_doe';

-- Резервное копирование и восстановление
-- В командной строке:
mysqldump -u username -p mydatabase > backup.sql
mysql -u username -p mydatabase < backup.sql

PostgreSQL

-- Подключение к серверу
psql -U username -d database

-- Создание базы данных
CREATE DATABASE mydatabase;
\c mydatabase

-- Создание таблицы
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Вставка данных
INSERT INTO users (username, email) VALUES ('john_doe', 'john@example.com');

-- Выборка данных с JOIN
SELECT u.username, p.title 
FROM users u
JOIN posts p ON u.id = p.user_id
WHERE u.username = 'john_doe';

-- Резервное копирование и восстановление
-- В командной строке:
pg_dump -U username mydatabase > backup.sql
psql -U username mydatabase < backup.sql

MongoDB

// Подключение к серверу
mongo

// Выбор базы данных
use mydatabase

// Создание коллекции и вставка документа
db.users.insertOne({
    username: "john_doe",
    email: "john@example.com",
    created_at: new Date()
})

// Поиск документов
db.users.find({ username: "john_doe" })

// Обновление документа
db.users.updateOne(
    { username: "john_doe" },
    { $set: { email: "new_email@example.com" } }
)

// Удаление документа
db.users.deleteOne({ username: "john_doe" })

// Агрегация
db.users.aggregate([
    { $match: { active: true } },
    { $group: { _id: "$role", count: { $sum: 1 } } }
])
